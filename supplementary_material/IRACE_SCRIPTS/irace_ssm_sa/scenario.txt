## irace scenario for SSM-SA hyperparameter tuning.
## Run from inside irace_ssm_sa/ directory:
##   cd irace_ssm_sa && Rscript -e "library(irace); irace.cmdline()"
##
## IMPORTANT: Each SSM-SA evaluation takes 15–1850 s depending on the instance.
## Budget of 500 experiments ≈ ~200,000 s (~2-3 machine-days) is a deliberate
## trade-off to produce publishable irace results within a feasible wall-clock
## time. The MILP-limited instance (H_O2_#3_2p) will produce identical costs
## for all configurations — this is a structural finding, not an irace failure.

# Training instances file (one absolute path per line)
trainInstancesFile = "./instances.txt"

# Target runner script (must be executable)
targetRunner = "./target-runner"

# Parameter space definition
parameterFile = "./parameters.txt"

# Budget: 500 experiments (~2-3 machine-days given SSM-SA per-run cost)
# Full irace default (10,000) would require ~42 machine-days — infeasible.
maxExperiments = 500

# Reproducibility seed
seed = 12345

# Parallelism: 0 = auto-detect all available cores
parallel = 0

# Output log file
logFile = "./irace.Rdata"

# irace verbosity (0=quiet, 1=normal, 2=verbose, 3=debug)
debugLevel = 1
